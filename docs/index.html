// Define your schema
User as Node {
  id: ID<User>
  name: NaturalLanguage
} & OutboundEdges {
  todos: Edge<Todo.ownerId>
}

Todo as Node {
  id: ID<Todo>
  text: NaturalLanguage
  completed: Timestamp | null
  ownerId: ID<User>
}



// Query your data
function TodoList({viewer}) {
  const todos = useLiveQuery(
    () => viewer.queryTodos().whereCompleted(P.equals(null))
  );
  return (
    <li>
      {todos.map(todo => <Todo todo={todo} />)}
    </li>
  );
}
